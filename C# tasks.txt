public class Person
{

}
else
{
_age = value;

}
}

public class StudentSearcher : ISearchable
{
public void Search(List<Student> students, string keyword)
{
var matches = students.Where(s => s.Name.ToLower().Contains(keyword.ToLower())).ToList();
Console.WriteLine(`$"Found {matches.Count} results:");
foreach (var match in matches)
{
match.DisplayInfo();
}
}
}
